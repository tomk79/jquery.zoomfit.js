サンプルソース

基本の処理

<script type="text/javascript">
	$(window).load(function(){
		$.zoomfit
			.init({
				defWidth: 500,//デフォルトの基準値
				maxZoom:1//最大拡大率
			})
			.addElements('.zoomfit')
		;
	});
</script>
<div class="zoomfit">
	<!-- フィットさせるコンテンツ -->
</div>

.addElements() の第2引数による基準値の変更

<script type="text/javascript">
	$(window).load(function(){
		$.zoomfit
			.init(500,1)
			.addElements('.zoomfit')
			.addElements('.zoomfit400', 400)//第2引数で基準値を上書きできる
		;
	});
</script>
<div class="zoomfit400">
	<!-- フィットさせるコンテンツ -->
</div>

属性 data-std-width による基準値の変更

<div class="zoomfit" data-std-width="700">
	<!-- フィットさせるコンテンツ -->
	<!-- data-std-width による指定が最も強い -->
</div>
